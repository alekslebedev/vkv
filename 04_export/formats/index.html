vkv

Version v0.2.0

the swiss army knife when working with Vault KVv2 engines

output formats

base

vkv export -p secret -f=base               
secret/
├── v1: admin [key=value]
│   └── sub=********
├── v1: demo
│   └── foo=***
└── sub/
    ├── v1: demo
    │   ├── demo=***********
    │   ├── password=******
    │   └── user=*****
    └── sub2
        └── v2: demo [admin=false key=value]
            ├── admin=***
            ├── foo=***
            ├── password=********
            └── user=****

yaml

 vkv export -p secret -f=yaml                       
secret/:
  admin:
    sub: '********'
  demo:
    foo: '***'
  sub/:
    demo:
      demo: '***********'
      password: '******'
      user: '*****'
    sub2/:
      demo:
        admin: '***'
        foo: '***'
        password: '********'
        user: '****'

json

vkv export -p secret -f=json
{
  "secret/": {
    "admin": {
      "sub": "********"
    },

      "foo": "***"
    },
    "sub/": {
      "demo": {
        "demo": "***********",
        "password": "******",
        "user": "*****"
      },
      "sub2/": {
        "demo": {
          "admin": "***",
          "foo": "***",
          "password": "********",
          "user": "****"
        }
      }
    }
  }
}

export

vkv export -p secret -f=export
export admin='key'
export demo='hello world'
export foo='bar'
export password='password'
export sub='password'

eval $(vkv export -p secret -f=export)
echo $admin
key

policy

 vkv export -p secret -f=policy 
PATH                    CREATE  READ    UPDATE  DELETE  LIST    ROOT
secret/sub/sub2/demo    ✖       ✖       ✖       ✖       ✖       ✔
secret/admin            ✖       ✖       ✖       ✖       ✖       ✔
secret/demo             ✖       ✖       ✖       ✖       ✖       ✔
secret/sub/demo         ✖       ✖       ✖       ✖       ✖       ✔
